nyc_inspec = rest_inspec %>%
  janitor::clean_names() %>% 
  mutate(grade = as.factor(grade),
         grade = fct_relevel(grade, c("A", "B", "C", "P", "Z", "Not Yet Graded"))) %>% 
  mutate(boro = str_to_lower(boro),
         boro = as.factor(boro),
         boro = fct_relevel(boro, c("manhattan", "bronx", "brooklyn", "queens", "staten island")))
nyc_cusine_manh = nyc_inspec %>% 
  filter (boro == "manhattan", grade == "C") %>% 
  count(cuisine_description) %>% 
  mutate(cuisine_description = fct_reorder(cuisine_description, n)) %>% 
  filter(n >= 100)

  plot_ly(nyc_cusine_manh, x = ~cuisine_description, y = ~n, color = ~cuisine_description, type = "bar")
